/**
 * TextInputField Component
 * author: soorajc
 * created@: 17/07/2018
 */
 
import React, { Component } from 'react';
import {
  TextInput,
} from 'react-native';
import PropTypes from 'prop-types';
 
import Styles from './styles';
 
export default class TextInputField extends Component<> {
 
  constructor(props) {
    super(props);
    this.state = {
      text: props.defaultValue,
      error: false,
    };
  }
 
  formValidator(value) {
    this.setState({ text: value });
    const reg = this.props.regex;
    if (this.props.regex) {
      const placeholder = this.props.placeholder;
      if (reg.test(value)) {
        this.setState({ error: false });
        this.props.formErrorHandler(false, placeholder);
      } else {
        this.setState({ error: true });
        this.props.formErrorHandler(true, placeholder);
      }
    }
  }
 
  render() {
    const customStyles = this.state.error
    ?
    this.props.textInputErrorStyle
    :
    this.props.textInputStyle;
 
    const defaultStyles = this.state.error
    ?
    Styles.textInputErrorStyle
    :
    Styles.defaultInput;
 
    return (
      <TextInput
        testID={this.props.testID}
        multiline={this.props.multiline}
        numberOfLines={this.props.numberOfLines}
        editable={this.props.editable}
        ref={this.props.textInputRef}
        placeholder={this.props.placeholder}
        defaultValue={this.props.defaultValue}
        placeholderTextColor={this.props.placeholderTextColor}
        secureTextEntry={this.props.secureTextEntry}
        underlineColorAndroid={this.props.underlineColorAndroid}
        style={
          this.props.textInputStyle
          ?
           customStyles
          :
           defaultStyles
        }
        onChangeText={text => this.formValidator(text)}
        value={this.state.text}
      />
    );
  }
}
 
 
TextInputField.propTypes = {
  defaultValue: PropTypes.string,
  editable: PropTypes.bool,
  formErrorHandler: PropTypes.func,
  multiline: PropTypes.bool,
  numberOfLines: PropTypes.number,
  placeholder: TextInput.propTypes.placeholder,
  placeholderTextColor: TextInput.propTypes.placeholderTextColor,
  regex: PropTypes.shape(),
  secureTextEntry: TextInput.propTypes.secureTextEntry,
  testID: PropTypes.string,
  textInputErrorStyle: TextInput.propTypes.style,
  textInputRef: PropTypes.oneOfType([PropTypes.func, PropTypes.object]),
  textInputStyle: TextInput.propTypes.style,
  underlineColorAndroid: TextInput.propTypes.underlineColorAndroid,
};
 
TextInputField.defaultProps = {
  defaultValue: '',
  editable: true,
  errorMessage: '',
  formErrorHandler: '',
  labelErrorStyle: '',
  multiline: false,
  numberOfLines: 1,
  placeholder: '',
  placeholderTextColor: '',
  regex: '',
  secureTextEntry: false,
  testID: '',
  textInputErrorStyle: '',
  textInputRef: '',
  textInputStyle: Styles.defaultInput,
  underlineColorAndroid: '',
};
