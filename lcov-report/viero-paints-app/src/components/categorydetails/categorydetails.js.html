/**
 * Category Details Component
 * created @ 17/10/2018
 * This files contains the code for the Catgeory Details Component
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import * as Animatable from 'react-native-animatable';
import {
  SafeAreaView,
  ScrollView,
  TouchableOpacity,
  Dimensions,
  Image,
  View,
  Text,
  FlatList,
  TextInput,
  Platform,
} from 'react-native';
 
import NavHeader from '../navheader/navheader';
import DemoProducts from './demoproducts';
import { MEDIA_URL } from '../../controller/apiconfig';
import Styles from './styles';
 
const { height, width } = Dimensions.get('window');
const Formats = ['1kg', '2kg', '3kg', '4kg', '5kg', '6kg', '7kg'];
const Colors = ['#e57373', '#2196f3', '#ff5252', '#f9a825', '#795548', '#3949ab', '#7b1fa2', '#e57373', '#2196f3', '#ff5252', '#f9a825', '#795548', '#3949ab', '#7b1fa2'];
 
 
export default class CategoryDetails extends React.Component {
 
  constructor() {
    super();
    this.state = {
      quantity: '',
      format: '1kg',
      starRating: 5,
      dataLoaded: true,
      data: '',
      itemSelected: 0,
      item: DemoProducts[0],
      offset: 0,
      direction: 'right',
    };
    this.bounce = this.bounce.bind(this);
    this.handleScroll = this.handleScroll.bind(this);
  }
 
  setIndex(index, item) {
    // this.bounce();
    if (Platform.OS === 'ios') {
      this.listView.scrollTo({ x: index * 0.31 * height })
    } else {
      this.listView.scrollTo({ x: index * 0.31 * height })
    }
    this.setState({ itemSelected: index, item: item });
  }
 
  handleViewRef = ref => this.view = ref;
 
  bounce() {
    this.view.slideInLeft(500);
  }
 
  renderProduct(item, index) {
    return (
      <View>
        <TouchableOpacity
          onPress={() => this.setIndex(index, item)}
          style={this.state.itemSelected === index ? Styles.productCardActive : Styles.productCard}
        >
          <Image source={item.image} style={Styles.productCardImage} />
          {
            this.state.itemSelected === index
            ?
              <View />
            :
              <Text style={Styles.productCardTitle}>
                {item.name}
              </Text>
          }
        </TouchableOpacity>
      </View>
    );
  }
 
  handleFormat(item) {
    this.setState({ format: item });
  }
 
  handleScroll(event) {
    const currentOffset = event.nativeEvent.contentOffset.x;
    const direction = currentOffset > this.state.offset ? 'right' : 'left';
    this.setState({ offset: currentOffset, direction: direction });
    //console.log("the scroll data is ----->", direction);
  }
 
  renderColors(item) {
    return (
      <TouchableOpacity testID={item} style={Styles.colorTile}>
        <View style={[Styles.colorButton, { backgroundColor: item }]} />
        <Text style={Styles.colorLabel}>{item}</Text>
      </TouchableOpacity>
    )
  }
 
  render() {
    const { state } = this.props.navigation;
    const data = state.params.data;
    const imageUrl = state.params.imageUrl;
    const shortDescriptionIndex = data['custom_attributes'].findIndex(x => x.attribute_code === 'short_description');
    const shortDescription = data.custom_attributes[shortDescriptionIndex].value.replace(/<\/?[^>]+(>|$)/g, '');
    const descriptionIndex = data['custom_attributes'].findIndex(x => x.attribute_code === 'description');
    const description = data.custom_attributes[descriptionIndex].value.replace(/<\/?[^>]+(>|$)/g, '');
    return (
      <SafeAreaView style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        <ScrollView testID={'scrollView'}>
          <View
            style={
            [
              Styles.titleView,
              { height: shortDescription.length > 25
                ?
                 (0.15 * height) + (shortDescription.length / 10)
                 :
                 (0.15 * height),
              },
            ]
           }
          >
            <Text style={Styles.shortDescription}>
              {shortDescription}
            </Text>
            <Text style={Styles.title}>
              {data.name}
            </Text>
          </View>
          <View style={Styles.productImageView}>
            <Image source={{ uri: imageUrl }} style={Styles.productImage} />
          </View>
          <View style={Styles.configureTitleView}>
            <Text style={Styles.configureTitle}>
             Vuoi ottenere questo effeto?
            </Text>
            <Text style={Styles.configureSubTitle}>
             CONFIGURA I PRODOTTI E ACQUISTA IL SET
            </Text>
          </View>
          <ScrollView ref={ref => this.listView = ref} horizontal onScroll={this.handleScroll}>
            {
              DemoProducts.map((item, index) => this.renderProduct(item, index))
            }
          </ScrollView>
          <View ref={this.handleViewRef} style={Styles.detailedView}>
            <View style={Styles.detailedTitleView}>
              <Text style={Styles.detailedTitle}>
                {this.state.item.name}
              </Text>
            </View>
            <View style={Styles.seperator} />
            <View style={Styles.attributeView}>
              <View style={Styles.numberView}>
                <View style={Styles.inputView}>
                  <Text style={Styles.quantityLabel}>
                   Quantità
                  </Text>
                  <TextInput
                    style={Styles.quantityInput}
                    value={this.state.quantity}
                    onChangeText={text => this.setState({ quantity: text })}
                    underlineColorAndroid="transparent"
                    placeholder="01"
                    placeholderTextColor="black"
                  />
                </View>
                <View style={Styles.formatView}>
                  <Text style={Styles.quantityLabel}>
                   Formato
                  </Text>
                  <View style={Styles.formatMenu}>
                    <ScrollView horizontal>
                      {
                       Formats.map(item =>
                         <TouchableOpacity
                           onPress={() => this.handleFormat(item)}
                           style={
                             this.state.format === item
                             ?
                             Styles.formatButtonActive
                             :
                             Styles.formatButton
                           }
                         >
                           <Text>{item}</Text>
                         </TouchableOpacity>
                       )
                     }
                    </ScrollView>
                  </View>
                </View>
              </View>
            </View>
            <View style={Styles.colorView}>
              <Text style={Styles.selectColorTitle}>
                {'Personalizza Colore >' }
              </Text>
              <FlatList
                data={Colors}
                numColumns={5}
                contentContainerStyle={Styles.colorList}
                renderItem={({ item }) => this.renderColors(item)}
              />
            </View>
            <View style={Styles.thumbnailView}>
              <Image source={{ uri: imageUrl }} style={Styles.thumbnail} />
            </View>
            <View style={Styles.descriptionCard}>
              <View style={Styles.descriptionView}>
                <Text style={Styles.descriptionLabel}>
                  {description}
                </Text>
              </View>
              {/*
              <TouchableOpacity
                style={[Styles.loadButton,
                  { marginTop: Platform.OS === 'ios' ? (description.length / 1700) * height : (description.length / 2000) * height }
                ]}
              >
                <Text style={Styles.loadLabel}>
                  {"SCOPRI DI PIU'"}
                </Text>
              </TouchableOpacity>
            */}
            </View>
          </View>
        </ScrollView>
      </SafeAreaView>
    );
  }
}
 
CategoryDetails.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    state: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
