/**
 * Wish List
 * created @ 17/10/2018
 * This files contains the code for the Related Products
 */
 
import React from 'react';
 
import {
  View,
  TouchableOpacity,
  Text,
  Image,
  ActivityIndicator,
  FlatList,
} from 'react-native';
import { WISHLIST_URL, MEDIA_URL } from '../../controller/apiconfig';
import NavHeader from '../navheader/navheader';
import { apiCallHandler, getToken, getLocalData } from '../../controller/apicontroller';
 
import Styles from './styles';
 
const dummyImage = require('../../assets/product.png');
 
 
export default class RelatedProducts extends React.Component {
 
  constructor() {
    super();
    this.state = {
      productList: '',
      dataLoaded: false,
    };
  }
 
  componentDidMount() {
    getLocalData('userData').then(
      userData => this.loadCustomerData(userData),
    );
  }
 
  loadCustomerData(data) {
    const userData = JSON.parse(data);
    const customerId = userData.id;
    getToken().then(token => this.loadProducts(token, customerId));
  }
 
  loadProducts(token, customerId) {
    const formData = new FormData();
    formData.append('customerId', customerId);
    const body = formData;
    apiCallHandler('POST', WISHLIST_URL, token, body, 'multipart/form-data').then(
      data => this.processProducts(data),
    );
  }
 
  processProducts(data) {
    this.setState({ dataLoaded: true, productList: data });
  }
 
  showProductDetails(item) {
    this.props.navigation.navigate('ProductDetails', { data: item });
  }
 
  renderProductCard(item) {
    return (
      <TouchableOpacity
        style={Styles.productView}
        onPress={() => this.showProductDetails(item)}
      >
        {
           item.details.image !== null
          ?
            <Image source={{ uri: MEDIA_URL + item.details.image }} style={Styles.productImage} />
          :
            <Image source={dummyImage} style={Styles.productImage} />
        }
        <View style={Styles.productTitleView}>
          <Text numberOfLines={1} style={Styles.productTitle}>{item.name} </Text>
          <Text numberOfLines={1} style={Styles.productDescription}>{item.details.short_description.replace(/<\/?[^>]+(>|$)/g, '')}</Text>
        </View>
      </TouchableOpacity>
    );
  }
 
  render() {
    return (
      <View style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        <View style={Styles.titleView}>
          {
            this.state.dataLoaded
            &&
            <Text style={Styles.relatedProductsTitle}>
              LA MIA LISTA DESIDERI:
            </Text>
          }
        </View>
        {
          this.state.dataLoaded
           ?
            this.state.productList.length > 0
             ?
               <FlatList
                 extraData={this.state}
                 removeClippedSubviews
                 data={this.state.productList}
                 renderItem={({ item }) => this.renderProductCard(item)}
               />
             :
               <View style={Styles.emptyView}>
                 <Text style={Styles.emptyLabel}>
                   No Products To Show
                 </Text>
               </View>
          :
               <View style={Styles.loaderView}>
                 <ActivityIndicator size="large" color="#5F7070" />
               </View>
        }
      </View>
    );
  }
}
