/**
 * Lavori Details Component
 * created @ 17/10/2018
 * This files contains the code for the Lavori Details Component
 */
 
import React from 'react';
import Icon from 'react-native-vector-icons/FontAwesome';
import PropTypes from 'prop-types';
import Swiper from 'react-native-swiper';
import {
  SafeAreaView,
  View,
  TextInput,
  Image,
  TouchableOpacity,
  ActivityIndicator,
  ImageBackground,
  ScrollView,
  FlatList,
  Text,
} from 'react-native';
import NavHeader from '../navheader/navheader';
import DemoProducts from './demolavori';
import Styles from './styles';
 
const filterLogo = require('../../assets/filter.png');
const authorLogo = require('../../assets/author.png');
const viewIcon = require('../../assets/eye.png');
 
const Images = [require('../../assets/lavori1.png'), require('../../assets/lavori2.png'), require('../../assets/lavori3.png'), require('../../assets/lavori4.png')]
 
 
export default class LavoriDetails extends React.Component {
 
  constructor() {
    super();
    this.state = {
      dataLoaded: true,
      title: 'Tutti i lavori',
    };
  }
 
  renderProductCard(item) {
    const { state } = this.props.navigation;
    const data = state.params.data;
    return (
      <TouchableOpacity style={Styles.productCard}>
        <View style={Styles.productImageContainer}>
          <ImageBackground resizeMode="stretch" source={item.image} style={Styles.productCardImage}>
            <View style={Styles.overlayView}>
              <View style={Styles.authorView}>
                <View style={Styles.authorImageView}>
                  <Image source={authorLogo} style={Styles.authorImage} />
                </View>
                <View style={Styles.authorDetailsView}>
                  <Text style={Styles.authorLabel}>
                    {data.name}
                  </Text>
                  <Text style={Styles.descriptionLabel}>
                    {data.author + ' / ' + data.place}
                  </Text>
                </View>
                <View style={Styles.iconView}>
                  <Image source={viewIcon} style={Styles.icon} />
                </View>
              </View>
            </View>
          </ImageBackground>
        </View>
      </TouchableOpacity>
    )
  }
 
 
  render() {
    const { state } = this.props.navigation;
    const data = state.params.data;
    return (
      <SafeAreaView testID={'LavoriDetails'} style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        <ScrollView>
          <View style={Styles.carouselView}>
            <Swiper
              autoplay
              loop
              dotStyle={Styles.dotStyle}
              dot={<View style={Styles.dot} />}
              activeDot={<View style={Styles.activeDot} />}
              style={Styles.wrapper}
              showsButtons={false}
              paginationStyle={Styles.pagination}
            >
              {
               Images.map(item => <Image source={item} style={Styles.productImage} />)
              }
            </Swiper>
            <View style={Styles.titleView}>
              <Text style={Styles.name}>
                {data.name}
              </Text>
            </View>
          </View>
          <View style={Styles.propertyView}>
            <View style={[Styles.propertyCard, { borderBottomWidth: 1 }]}>
              <View style={Styles.labelView}>
                <Text style={Styles.label}>
                  Applicatore:
                </Text>
              </View>
              <View style={Styles.valueView}>
                <Text style={Styles.value}>
                  {data.author}
                </Text>
              </View>
            </View>
          </View>
          <View style={Styles.propertyView}>
            <View style={[Styles.propertyCard, { borderBottomWidth: 1 }]}>
              <View style={Styles.labelView}>
                <Text style={Styles.label}>
                  Zone:
                </Text>
              </View>
              <View style={Styles.valueView}>
                <Text style={Styles.value}>
                  {data.place}
                </Text>
              </View>
            </View>
          </View>
          <View style={Styles.propertyView}>
            <View style={Styles.propertyCard}>
              <View style={Styles.labelView}>
                <Text style={Styles.label}>
                  Prodotti:
                </Text>
              </View>
              <View style={Styles.valueView}>
                <Text style={[Styles.value, { textDecorationLine: 'underline', textDecorationColor: '#5F7070', textDecorationStyle: 'solid' }]}>
                  Allure
                </Text>
              </View>
            </View>
          </View>
          <View style={Styles.descriptionView}>
            <Text style={Styles.description}>
              L’effetto decorativo Marmorin è stato scelto per decorare il muro lungo la piscina,
              la palestra e l’area lounge di una villa privata nel Regno Unito.
            </Text>
          </View>
          <View style={Styles.productList}>
            <Text style={Styles.productListTitle}>
              {'ALTRI LAVORI DI ' + data.author.toUpperCase() + ':'}
            </Text>
            {
              DemoProducts.map(item => this.renderProductCard(item))
            }
          </View>
          <View style={Styles.buttonView}>
            <TouchableOpacity style={Styles.button}>
              <Text style={Styles.buttonLabel}>
                Visualizza tutti i progetti
              </Text>
            </TouchableOpacity>
          </View>
        </ScrollView>
      </SafeAreaView>
    );
  }
}
 
LavoriDetails.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
