/**
 * PublishLavori View
 * created @ 17/10/2018
 * This files contains the code for the PublishLavori View
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import { KeyboardAwareScrollView } from 'react-native-keyboard-aware-scroll-view'
import {
  SafeAreaView,
  View,
  Text,
  Image,
  TouchableOpacity,
  Dimensions,
} from 'react-native';
 
import NavHeader from '../navheader/navheader';
import TextInputField from '../../lib/textinput/textinput';
import Styles from './styles';
 
const { height, width } = Dimensions.get('window');
const buttonImage = require('../../assets/button.png');
const plusButton = require('../../assets/uploadbutton.png');
 
export default class PublishLavori extends React.Component {
 
  renderInput(label, defaultValue, lines){
    return (
      <View style={Styles.inputCard}>
        <View style={Styles.labelView}>
          <Text style={Styles.inputLabel}>
            {label}
          </Text>
        </View>
        <View style={label === 'Testo :' ? [Styles.inputView, { height: 0.11 * height, borderRadius: 0.05 * width }] : Styles.inputView}>
          <TextInputField
            textInputRef={(arg) => { this.name = arg; }}
            defaultValue={defaultValue}
            numberOfLines={lines}
            multiline
            underlineColorAndroid="transparent"
            textInputStyle={Styles.textInput}
          />
        </View>
      </View>
    )
  }
 
  renderAuthor(name) {
    return (
      <View style={Styles.nameCard}>
        <View style={Styles.nameLabelView}>
          <Text style={Styles.inputLabel}>
            Applicatore:
          </Text>
        </View>
        <View>
          <Text style={Styles.authorName}>
            {name}
          </Text>
        </View>
      </View>
    )
  }
 
  render() {
    return (
      <SafeAreaView testID={'PublishLavori'} style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        <KeyboardAwareScrollView>
        <TouchableOpacity style={Styles.uploadView}>
          <Image source={plusButton} style={Styles.plusButton} />
          <Text style={Styles.uploadLabel}>
            Puoi caricare fino ad un massimo di 3 foto.
            L'utente dichiara di essere il proprietario delle foto inserite e l'effettivo esecutore dei lavori svolti.
          </Text>
        </TouchableOpacity>
        {this.renderInput('Titolo :', 'MAX LENGTH 25', 1)}
        {this.renderInput('Testo :', 'MAX LENGTH 60', 10)}
        {this.renderAuthor('Mario Rossi')}
        {this.renderInput('Zona :', '', 1)}
        {this.renderInput('Prodotti :', '', 1)}
        <TouchableOpacity style={Styles.publishButton}>
          <Text style={Styles.publishLabel}>
            PUBBLICA
          </Text>
        </TouchableOpacity>
        </KeyboardAwareScrollView>
      </SafeAreaView>
    );
  }
}
 
PublishLavori.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    state: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
