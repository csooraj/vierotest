/**
 * Profile View
 * created @ 17/10/2018
 * This files contains the code for the Profile View
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import {
  SafeAreaView,
  View,
  Image,
  TouchableOpacity,
  Text,
  ScrollView,
} from 'react-native';
 
import NavHeader from '../navheader/navheader';
import Styles from './styles';
 
const infoImage = require('../../assets/profile.png');
 
const MenuItems = [{ label: 'Informazioni Account', key: 'EditProfile' }, { label: 'I mie Ordini', key: 'Orders' }, { label: 'La mia Lista desideri', key: '' }, { label: 'Area Download', key: 'Education' }];
 
export default class Profile extends React.Component {
 
  handleMenu(key) {
    const { state } = this.props.navigation;
    const userData = state.params.data;
    const name = (userData.firstname + ' ' + userData.lastname).toUpperCase();
    const data = { name: name, details: userData };
    this.props.navigation.navigate(key, { data: data });
  }
 
  render() {
    const { state } = this.props.navigation;
    const userData = state.params.data;
    return (
      <SafeAreaView style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        <ScrollView contentContainerStyle={Styles.scrollView}>
          <View style={Styles.imageView}>
            <Image source={infoImage} style={Styles.profileImage} />
          </View>
          <View style={Styles.titleView}>
            <Text style={Styles.title}>
              {userData.firstname + ' ' + userData.lastname}
            </Text>
            <Text style={Styles.subTitle}>
              Decoratore Professionista
            </Text>
          </View>
          {
            MenuItems.map(item =>
              <TouchableOpacity
                testID={item.key}
                accessibilityLabel={item.key}
                onPress={() => this.handleMenu(item.key)}
                style={Styles.menuButton}
              >
                <Text style={Styles.menuLabel}>
                  {item.label}
                </Text>
              </TouchableOpacity>,
            )
          }
        </ScrollView>
      </SafeAreaView>
    );
  }
}
 
Profile.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    state: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
