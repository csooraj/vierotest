/**
 * Login Screen Component
 * created @ 17/10/2018
 * This files contains the code for the Login Screen Component
 */
 
import React, { Component } from 'react';
import { StackActions, NavigationActions } from 'react-navigation';
import {
  View,
  ImageBackground,
  Image,
  Text,
  TouchableOpacity,
  Alert,
  ActivityIndicator,
  AsyncStorage,
} from 'react-native';
 
import Styles from './styles';
import { LOGIN_URL, PROFILE_URL } from '../../controller/apiconfig';
import { apiCallHandler } from '../../controller/apicontroller';
import TextInputField from '../../lib/textinput/textinput';
import FormFields from './formconfig';
 
const LOGIN_BACKGROUND = require('../../assets/splash.png');
const LOGO = require('../../assets/logo.png');
 
 
const addedFields = [];
let errorStatus = [];
 
export default class Login extends Component<{}> {
 
  constructor() {
    super();
    this.formErrorHandler = this.formErrorHandler.bind(this);
    this.state = {
      loading: false,
      enableLoginButton: false,
    };
  }
 
  formErrorHandler(error, fieldName) {
    addedFields.push(fieldName);
    const fieldSet = new Set(addedFields);
 
    if (error) {
      errorStatus.push(fieldName);
      this.setState({ enableLoginButton: false });
    } else {
      errorStatus = errorStatus.filter((element) => {
        return element !== fieldName;
      });
      if ((fieldSet.size === FormFields.length) && (errorStatus.length === 0)) {
        this.setState({ enableLoginButton: true });
      } else {
        this.setState({ enableLoginButton: false });
      }
    }
  }
 
  handleLogin() {
    this.setState({ loading: true });
    const username = this.email._lastNativeText;
    const password = this.password._lastNativeText;
    const userData = new FormData();
    userData.append('password', password);
    userData.append('username', username);
    console.log("The username and password", username, password);
    apiCallHandler('POST', LOGIN_URL, '', userData, 'multipart/form-data').then(
      data => this.loginResultHandler(data));
  }
 
  loginResultHandler(data) {
    if (data.hasOwnProperty('message')) {
      this.setState({ loading: false });
      Alert.alert('Invalid Username or Password Try Again');
    } else {
      this.props.loginSuccessCallBack();
      AsyncStorage.setItem('token', data);
      this.loadCustomerDetails(data);
    }
  }
 
  loadCustomerDetails(token) {
    apiCallHandler('GET', PROFILE_URL, token, '', 'application/json').then(
      data => this.profileDataHandler(data));
  }
 
  profileDataHandler(data) {
    console.log("The data before the async storage", data);
    AsyncStorage.setItem('userData', JSON.stringify(data));
    this.goToHomeScreen();
  }
 
  goToHomeScreen() {
    const resetAction = StackActions.reset({
      index: 0,
      actions: [NavigationActions.navigate({ routeName: 'Home' })],
    });
    this.props.navigation.dispatch(resetAction);
  }
 
  render() {
    return (
      <View style={Styles.container} >
        <ImageBackground
          testID={'LoginImage'}
          resizeMode="cover"
          source={LOGIN_BACKGROUND}
          style={Styles.imageBackground}
        >
          <View style={Styles.contentView}>
            <View style={Styles.logoContainer}>
              <Image
                style={Styles.logo}
                testID={'AppLogo'}
                source={LOGO}
              />
            </View>
            <View style={Styles.formContainer}>
              {
               FormFields.map(item =>
                 <TextInputField
                   testID={item.testID}
                   key={item.key}
                   formErrorHandler={this.formErrorHandler}
                   textInputRef={(arg) => { this[`${item.ref}`] = arg; }}
                   placeholder={item.placeholder}
                   secureTextEntry={item.secureTextEntry}
                   regex={item.regex}
                   errorMessage={item.errorMessage}
                   placeholderTextColor="#FFFFFF"
                   underlineColorAndroid="transparent"
                   textInputStyle={Styles.textInput}
                   marginBottom={1}
                   textInputErrorStyle={Styles.textInputErrorStyle}
                 />,
               )
             }
              {
               this.state.loading
               ?
                 <View style={Styles.loginButton}>
                   <ActivityIndicator size="large" color="#FFFFFF" />
                 </View>
               :
                 <TouchableOpacity
                   disabled={!this.state.enableLoginButton}
                   onPress={() => this.handleLogin()}
                   testID={'LoginButton'}
                   accessibilityLabel={'LoginButton'}
                   style={Styles.loginButton}
                 >
                   <Text
                     style={
                      this.state.enableLoginButton
                      ?
                      Styles.loginLabel
                      :
                      Styles.passiveLabel
                    }
                   >
                     {'Accedi >'}
                   </Text>
                 </TouchableOpacity>
             }
            </View>
          </View>
        </ImageBackground>
      </View>
    );
  }
}
