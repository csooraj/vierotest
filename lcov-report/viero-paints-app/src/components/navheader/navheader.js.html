/**
 * NavHeader Component
 * created @ 17/10/2018
 * This files contains the code for the NavHeader Component
 */
 
import React from 'react';
import {
  Image,
  View,
  TouchableOpacity,
} from 'react-native';
import Styles from './styles';
 
const menuBarImage = require('../../assets/bar.png');
const appLogo = require('../../assets/viero.png');
const backIcon = require('../../assets/back.png');
 
export default class NavHeader extends React.Component {
 
  showMenu() {
    this.props.navigation.navigate('Menu');
  }
 
  render() {
    return (
      <View style={Styles.header} testID={'NavHeader'}>
        <TouchableOpacity testID={'FrontButton'} style={Styles.frontView} onPress={() => this.props.navigation.goBack()}>
          {this.props.showBack && <Image source={backIcon} style={Styles.menuBarImage} />}
        </TouchableOpacity>
        <View style={Styles.middleView}>
          <Image source={appLogo} style={Styles.appLogo} />
        </View>
        <TouchableOpacity
          testID={'EndButton'}
          onPress={() => this.showMenu()}
          style={Styles.endView}
        >
          <Image source={menuBarImage} style={Styles.menuBarImage} />
        </TouchableOpacity>
      </View>
    );
  }
}
