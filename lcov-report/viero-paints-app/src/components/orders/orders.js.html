    const userId = state.params.data.details.id;
    const url = ORDER_LIST_URL + userId.toString() + ORDER_SORT_URL + pageNo;
    apiCallHandler('GET', url, token, '', 'application/json').then(

  loadMoreData(token) {
    const { state } = this.props.navigation;
    const userId = state.params.data.details.id;
    const url = ORDER_LIST_URL + userId.toString() + ORDER_SORT_URL + pageNo;

  refreshData(data) {
    const setData = this.state.orderList.concat(data.items);
