/**
 * Related Products
 * created @ 17/10/2018
 * This files contains the code for the Related Products
 */
 
import React from 'react';
 
import {
  View,
  TouchableOpacity,
  Text,
  Image,
  ActivityIndicator,
} from 'react-native';
import { PRODUCTTI_URL_HEAD, PRODUCTTI_URL_TAIL, MEDIA_URL } from '../../controller/apiconfig';
import { apiCallHandler, getToken } from '../../controller/apicontroller';
 
import Styles from './styles';
 
const dummyImage = require('../../assets/product.png');
 
 
export default class RelatedProducts extends React.Component {
 
  constructor() {
    super();
    this.state = {
      productList: '',
      dataLoaded: false,
    };
  }
 
  componentDidMount() {
    const url = PRODUCTTI_URL_HEAD + 4 + PRODUCTTI_URL_TAIL;
    getToken().then(token => this.loadProducts(token, url));
  }
 
  loadProducts(token, url) {
    apiCallHandler('GET', url, token, '', 'application/json').then(
      data => this.processProducts(data)
    );
  }
 
  processProducts(data) {
    this.setState({ productList: data.items, dataLoaded: true });
  }
 
  renderProductCard(item) {
    const imageIndex = item['custom_attributes'].findIndex(x => x.attribute_code === 'thumbnail');
    const descriptionIndex = item['custom_attributes'].findIndex(x => x.attribute_code === 'short_description');
    return (
      <TouchableOpacity
        onPress={() => this.props.showProductDetails(item)}
        style={Styles.productView}
      >
        {
           imageIndex >= 0
          ?
            <Image source={{ uri: MEDIA_URL + item.custom_attributes[imageIndex]['value'] }} style={Styles.productImage} />
          :
            <Image source={dummyImage} style={Styles.productImage} />
        }
        <View style={Styles.productTitleView}>
          <Text numberOfLines={1} style={Styles.productTitle}>{item.name} </Text>
          <Text numberOfLines={1} style={Styles.productDescription}>{item.custom_attributes[descriptionIndex]['value'].replace(/<\/?[^>]+(>|$)/g, '')}</Text>
        </View>
      </TouchableOpacity>
    );
  }
 
  render() {
    return (
      <View>
        <View style={Styles.titleView}>
          <Text style={Styles.relatedProductsTitle}>
            PRODOTTI CORRELATI:
          </Text>
        </View>
        {
          this.state.dataLoaded
          ?
          this.state.productList.map(item =>
            this.renderProductCard(item),
          )
          :
          <View style={Styles.loaderView}>
            <ActivityIndicator size="large" color="#5F7070" />
          </View>
        }
      </View>
    );
  }
}
