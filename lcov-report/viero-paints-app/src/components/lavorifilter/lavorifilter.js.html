/**
 * Lavori Filter Component
 * created @ 17/10/2018
 * This files contains the code for the LavoriFilter Component
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import {
  View,
  TouchableOpacity,
  Dimensions,
  ActivityIndicator,
  Text,
} from 'react-native';
import Styles from './styles';
 
const { height } = Dimensions.get('window');
 
export default class LavoriFilter extends React.Component {
 
  constructor() {
    super();
    this.state = {
      filterList: '',
      index: 1,
    };
  }
 
  componentWillMount() {
    this.setState({ filterList: this.props.filterData[0].Topology });
  }
 
  setIndex(index) {
    this.setState({ index: index });
    if (index === 1) {
      this.setState({ filterList: this.props.filterData[0].Topology });
    } else {
      this.setState({ filterList: this.props.filterData[1].Effecti });
    }
  }
 
  displayFilter() {
    if (this.state.filterList.length > 0) {
      const itemLength = this.state.filterList.length;
      return (
        <View style={Styles.filterContainer} testID={'LavoriFilter'}>
          <View style={Styles.buttonView}>
            <TouchableOpacity
              onPress={() => this.setIndex(1)}
              style={[Styles.button, { backgroundColor: this.state.index === 1 ? '#586767' : '#5F7070' }]}
            >
              <Text style={[Styles.buttonLabel, { fontFamily: this.state.index === 1 ? 'BrauerNeue-Bold' : 'BrauerNeue-Regular' }]}>
                TIPOLOGIA
              </Text>
            </TouchableOpacity>
            <TouchableOpacity
              onPress={() => this.setIndex(2)}
              style={[Styles.button, { backgroundColor: this.state.index === 2 ? '#586767' : '#5F7070' }]}
            >
              <Text style={[Styles.buttonLabel, { fontFamily: this.state.index === 2 ? 'BrauerNeue-Bold' : 'BrauerNeue-Regular' }]}>
                EFFETI
              </Text>
            </TouchableOpacity>
          </View>
          <View style={[Styles.filterView, { paddingBottom: '4.9%' }]} >
            {
             this.state.filterList.map((item, index) =>
               <TouchableOpacity
                 onPress={() => this.props.filterItems(item)}
                 style={
                   index === itemLength - 1
                   ?
                   [Styles.filterItem, { borderBottomWidth: 0 }]
                   :
                   Styles.filterItem
                 }
               >
                 <Text
                   style={
                      this.props.title === item
                      ?
                      Styles.filterItemLabelActive
                      :
                      Styles.filterItemLabel
                  }
                 >
                   {item}
                 </Text>
               </TouchableOpacity>
             )
           }
          </View>
        </View>
      )
    } else {
      return (
        <View style={Styles.loaderView} >
          <ActivityIndicator />
        </View>
      );
    }
  }
 
  render() {
    return (
        this.displayFilter()
    );
  }
}
 
LavoriFilter.propTypes = {
  title: PropTypes.string,
  filterData: PropTypes.arrayOf(),
  filterItems: PropTypes.func,
};
 
LavoriFilter.defaultProps = {
  title: '',
  filterData: '',
  filterItems: null,
};
