 * Informazioni View
 * created @ 17/10/2018
 * This files contains the code for the Informazioni View
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import HTMLView from 'react-native-htmlview';
import {
  SafeAreaView,
  ScrollView,
  Dimensions,
  View,
  Text,
  Image,
} from 'react-native';
 
import NavHeader from '../navheader/navheader';
import { INFO_URL } from '../../controller/apiconfig';
import { apiCallHandler, getToken } from '../../controller/apicontroller';
import Styles from './styles';
 
const infoImage = require('../../assets/info.png');
 
const { height } = Dimensions.get('window');
 
export default class Informazioni extends React.Component {
 
  constructor() {
    super();
    this.state = {
      content: '',
      dataLoaded: false,
    };
  }
 
  componentDidMount() {
    getToken().then(token => this.loadPageContent(token));
  }
 
  loadPageContent(token) {
    apiCallHandler('GET', INFO_URL, token, '', 'application/json').then(
      data => this.processContent(data),
    );
  }
 
  processContent(data) {
    this.setState({ content: data, dataLoaded: true });
  }
 
  render() {
    return (
      <SafeAreaView style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        <ScrollView contentContainerStyle={Styles.scrollView}>
          <View style={Styles.imageView}>
            <Image source={infoImage} style={Styles.infoImage} />
          </View>
          <Text style={Styles.infoTitle}>
            Informazioni Utili
          </Text>
          <View style={Styles.contentView}>
            <View style={Styles.headerView}>
              <Text style={Styles.headerLabel}>
                CONSEGNA ESPRESSA
              </Text>
            </View>
            <View style={Styles.descriptionView}>
              <Text style={Styles.description}>
                {
                  "Viero offre un servizio di consegna espressa. Il pacco verrà consegnato tramite corriere DHL entro 1-2 giorni dalla spedizione (dipende dal tuo CAP).
