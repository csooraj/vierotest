/**
 * Property View  Component
 * created @ 17/10/2018
 * This files contains the code for the Property View
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import {
  View,
  Text,
} from 'react-native';
 
import Styles from './styles';
 
 
export default class PropertyView extends React.Component {
 
  render() {
    return (
      <View style={[Styles.propertyCard, { backgroundColor: this.props.index % 2 === 0 ? '#F7F7F7' : 'white' }]}>
        <View style={Styles.propertyTitleView}>
          <Text style={Styles.propertyTitle}>
            {this.props.title + ':'}
          </Text>
        </View>
        <View style={Styles.propertyValueView}>
          <Text style={Styles.propertyLabel}>
            {this.props.value}
          </Text>
        </View>
      </View>
    );
  }
}
 
PropertyView.propTypes = {
  title: PropTypes.string,
  value: PropTypes.string,
  index: PropTypes.number,
};
 
PropertyView.defaultProps = {
  title: '',
  value: '',
  index: 0,
};
