/**
 * Product Details Component
 * created @ 17/10/2018
 * This files contains the code for the Product Details Component
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import StarRating from 'react-native-star-rating';
import HTMLView from 'react-native-htmlview';
import {
  SafeAreaView,
  ScrollView,
  TouchableOpacity,
  Dimensions,
  ActivityIndicator,
  Image,
  View,
  Text,
  FlatList,
  TextInput,
} from 'react-native';
 
import NavHeader from '../navheader/navheader';
import MaterialCard from './materialcard';
import { MEDIA_URL } from '../../controller/apiconfig';
import Styles from './styles';
import DescriptionStyles from './descriptionstyles';
import PropertyCard from './propertyview';
import AttachmentView from './attachmentview';
import RelatedProducts from '../relatedproducts/relatedproducts';
 
const { height } = Dimensions.get('window');
const Formats = ['1kg', '2kg', '3kg', '4kg', '5kg', '6kg', '7kg'];
const Colors = ['#e57373', '#2196f3', '#ff5252', '#f9a825', '#795548', '#3949ab', '#7b1fa2', '#e57373', '#2196f3', '#ff5252', '#f9a825', '#795548', '#3949ab', '#7b1fa2'];
const PropertyList = [{ name: 'TOOLS PRINCIPALI', value: 'Splater' }, { name: 'RESA E CONSUMO', value: '7.0- 8.0 m2/l' }, { name: 'CLASSE DE EMISSIONE', value: 'A+' }, { name: 'BASE', value: 'Base Calce' }];
const Attachments = ['SCHEDA TECNICA', 'SCHEDA DI SICUREZZA', 'COLLEZIONE COLORE'];
const materialImage = require('../../assets/material.png');
const colorBucketImage = require('../../assets/color.png');
 
export default class ProductDetails extends React.Component {
 
  constructor() {
    super();
    this.state = {
      quantity: '',
      format: '1kg',
      starRating: 5,
      dataLoaded: true,
      data: '',
    };
    this.showRelatedProductDetails = this.showRelatedProductDetails.bind(this);
  }
 
  componentWillMount() {
    const { state } = this.props.navigation;
    const data = state.params.data;
    this.setState({ data: data });
  }
 
  setRating(rating) {
    this.setState({ starRating: rating });
  }
 
 
  handleFormat(item) {
    this.setState({ format: item });
  }
 
  showRelatedProductDetails(item) {
    this.setState({ dataLoaded: false });
    this.setState({ data: item });
    setTimeout(() => this.hideLoader(), 200);
  }
 
  hideLoader() {
    this.setState({ dataLoaded: true });
  }
 
  renderColors(item) {
    return (
      <TouchableOpacity style={Styles.colorTile}>
        <View style={[Styles.colorButton, { backgroundColor: item }]} />
        <Text style={Styles.colorLabel}>{item}</Text>
      </TouchableOpacity>
    )
  }
 
  render() {
    const data = this.state.data;
    const imageIndex = data['custom_attributes'].findIndex(x => x.attribute_code === 'thumbnail');
    const descriptionIndex = data['custom_attributes'].findIndex(x => x.attribute_code === 'description');
    return (
      <SafeAreaView style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        {
          this.state.dataLoaded
          ?
            <ScrollView>
              <View style={Styles.productImageView}>
                <Image source={{ uri: MEDIA_URL + data.custom_attributes[imageIndex]['value'] }} style={Styles.productImage} />
              </View>
              <View
                style={[Styles.productMetaDataView,
                  {
                    height:
                     data.name.length > 15
                     ?
                    ((0.20 + (data.name.length / 1000)) * height)
                    :
                    0.20 * height
                  }]}
              >
                <View style={Styles.productTitleView}>
                  <Text style={Styles.productName}>
                    {data.name}
                  </Text>
                  <Text style={Styles.productPrice}>
                    {'€ 49.50'}
                  </Text>
                </View>
                <View style={Styles.productRatingView}>
                  <View style={Styles.starRatingView}>
                    <StarRating
                      disabled
                      emptyStarColor="#5B7171"
                      fullStarColor="#5B7171"
                      starSize={15}
                      maxStars={5}
                      rating={4}
                    />
                  </View>
                  <View style={Styles.buyButtonView}>
                    <TouchableOpacity style={Styles.acquistaButton}>
                      <Text style={Styles.acquistaLabel}>
                        ACQUISTA
                      </Text>
                    </TouchableOpacity>
                  </View>
                </View>
              </View>
              <View style={Styles.attributeView}>
                <View style={Styles.numberView}>
                  <View style={Styles.inputView}>
                    <Text style={Styles.quantityLabel}>
                     Quantità
                    </Text>
                    <TextInput
                      style={Styles.quantityInput}
                      value={this.state.quantity}
                      onChangeText={text => this.setState({ quantity: text })}
                      underlineColorAndroid="transparent"
                      placeholder="01"
                      placeholderTextColor="black"
                    />
                  </View>
                  <View style={Styles.formatView}>
                    <Text style={Styles.quantityLabel}>
                     Formato
                    </Text>
                    <View style={Styles.formatMenu}>
                      <ScrollView horizontal>
                        {
                         Formats.map(item =>
                           <TouchableOpacity
                             onPress={() => this.handleFormat(item)}
                             style={
                               this.state.format === item
                               ?
                               Styles.formatButtonActive
                               :
                               Styles.formatButton
                             }
                           >
                             <Text style={Styles.formatLabel}>{item}</Text>
                           </TouchableOpacity>
                         )
                       }
                      </ScrollView>
                    </View>
                  </View>
                </View>
              </View>
              <View style={Styles.colorView}>
                <Text style={Styles.selectColorTitle}>
                 Scegli il colore
                </Text>
                <FlatList
                  data={Colors}
                  numColumns={5}
                  contentContainerStyle={Styles.colorList}
                  renderItem={({ item }) => this.renderColors(item)}
                />
              </View>
              <View style={Styles.materialView}>
                <MaterialCard
                  image={materialImage}
                  title="Interni"
                  showBorder
                />
                <MaterialCard
                  image={colorBucketImage}
                  title="Sandy Metallic"
                />
              </View>
              <View style={Styles.descriptionView}>
                <HTMLView
                  value={data.custom_attributes[descriptionIndex].value}
                  stylesheet={DescriptionStyles}
                />
              </View>
              <View style={Styles.propertyList}>
                {
                 PropertyList.map((item, index) =>
                   <PropertyCard
                     index={index}
                     title={item.name}
                     value={item.value}
                   />,
                 )
               }
              </View>
              <View style={Styles.attachmentList}>
                {
                 Attachments.map(item =>
                   <AttachmentView
                     title={item}
                   />,
                 )
                }
              </View>
              <View style={Styles.ratingButtonView}>
                <Text style={Styles.ratingLabel}>
                  Dai una valutazione a questo prodotto
                </Text>
                <View style={Styles.ratingButton}>
                  <StarRating
                    disabled={false}
                    emptyStarColor="#5B7171"
                    fullStarColor="#5B7171"
                    starSize={26}
                    maxStars={5}
                    rating={this.state.starRating}
                    animation="bounce"
                    selectedStar={rating => this.setRating(rating)}
                  />
                </View>
              </View>
              <View style={Styles.relatedProductsView}>
                <RelatedProducts
                  showProductDetails={this.showRelatedProductDetails}
                />
              </View>
            </ScrollView>
          :
            <View style={Styles.loaderView}>
              <ActivityIndicator size="large" color="#4D4C4C" />
            </View>
        }
      </SafeAreaView>
    );
  }
}
 
ProductDetails.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    state: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
