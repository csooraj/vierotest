/**
 * Material Card Component
 * created @ 17/10/2018
 * This files contains the code for the Material Card
 */
 
import React from 'react';
import PropTypes from 'prop-types';
import {
  Image,
  View,
  Text,
} from 'react-native';
 
import Styles from './styles';
 
 
export default class MaterialCard extends React.Component {
 
  render() {
    return (
      <View style={Styles.materialCard}>
        <View
          style={
            this.props.showBorder
            ?
            Styles.materialImageViewWithBorder
            :
            Styles.materialImageView
          }
        >
          <Image source={this.props.image} style={Styles.materialImage} />
        </View>
        <View
          style={
            this.props.showBorder
            ?
            Styles.materialLabelViewWithBorder
            :
            Styles.materialLabelView
          }
        >
          <Text style={Styles.materialLabel}>
            {this.props.title}
          </Text>
        </View>
      </View>
    );
  }
}
 
MaterialCard.propTypes = {
  image: Image.propTypes.source,
  title: PropTypes.string,
  showBorder: PropTypes.boolean,
};
 
MaterialCard.defaultProps = {
  image: '',
  title: '',
  showBorder: false,
};
