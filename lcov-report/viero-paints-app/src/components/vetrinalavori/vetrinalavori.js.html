/**
 * Product List Component
 * created @ 17/10/2018
 * This files contains the code for the Product Component
 */
 
import React from 'react';
import Icon from 'react-native-vector-icons/FontAwesome';
import PropTypes from 'prop-types';
import {
  SafeAreaView,
  View,
  TextInput,
  Image,
  TouchableOpacity,
  ActivityIndicator,
  ImageBackground,
  ScrollView,
  FlatList,
  Text,
} from 'react-native';
import NavHeader from '../navheader/navheader';
import LavoriFilter from '../lavorifilter/lavorifilter';
import Styles from './styles';
import DemoProducts from './demolavori';
 
const filterLogo = require('../../assets/filter.png');
const tagIcon = require('../../assets/tag.png');
const filterLabels = [{ Topology: ['Tutti i prodotti', 'Prymer & Fondi', 'Finiture', 'Additivi Decorativi', 'Protezioni'] }, { Effecti: ['Tutti Effect', 'Metallica', 'Wood', 'Plain', 'Material'] }];
 
export default class VetrinaLavori extends React.Component {
 
  constructor() {
    super();
    this.state = {
      productList: DemoProducts,
      filteredFullData: DemoProducts,
      dataLoaded: true,
      title: 'Tutti i lavori',
      showFilter: false,
    };
    this.filterItems = this.filterItems.bind(this);
  }
 
  showDetails(item) {
    this.props.navigation.navigate('LavoriDetails', { data: item });
  }
 
  searchItems(keyword) {
    this.setState({ searchTerm: keyword });
    const data = this.state.productList;
    const filteredList = data.filter((item) => {
      const matchIndex = item.name.toLowerCase().search(keyword.toLowerCase());
      if (matchIndex !== -1) {
        return item;
      }
    });
    if (filteredList.length > 0) {
      this.setState({ productList: filteredList });
    } else {
      this.setState({ productList: data });
    }
    if (keyword === '') {
      if (this.state.filterActive) {
        const fullFilteredData = this.state.filteredFullData;
        this.setState({ productList: fullFilteredData });
      } else {
        this.setState({ productList: DemoProducts });
      }
    }
  }
 
  renderProductCard(item) {
    return (
      <TouchableOpacity
        onPress={() => this.showDetails(item)}
        style={Styles.productCard}
      >
        <View style={Styles.productImageContainer}>
          <ImageBackground resizeMode="stretch" source={item.image} style={Styles.productImage}>
            <View style={Styles.overlayView}>
              <View style={Styles.paddedView} />
              <View style={Styles.tagView}>
                <View style={Styles.tagIconContainer}>
                  <Image source={tagIcon} style={Styles.tagIcon} />
                </View>
                <View style={Styles.tagContainer}>
                  <Text style={Styles.tagLabel}>
                    {item.tag}
                  </Text>
                </View>
              </View>
            </View>
          </ImageBackground>
        </View>
        <View style={Styles.productTitleView}>
          <Text style={Styles.productTitle}>{item.name} </Text>
          <Text style={Styles.productDescription}>{item.author + ' / ' + item.place}</Text>
        </View>
      </TouchableOpacity>
    )
  }
 
  showFilter() {
    if (this.state.showFilter) {
      this.setState({ showFilter: false });
    } else {
      this.setState({ showFilter: true });
    }
  }
 
  displayFilter() {
    if (this.state.showFilter) {
      return (
        <LavoriFilter
          filterData={filterLabels}
          filterItems={this.filterItems}
          title={this.state.title}
        />
      )
    } else {
      return (null);
    }
  }
 
  filterItems(item) {
    const title = item;
    this.setState({ title: title, showFilter: false });
  }
 
  addNewItem() {
    this.props.navigation.navigate('PublishLavori');
  }
 
 
  render() {
    return (
      <SafeAreaView testID={'Lavori'} style={Styles.container}>
        <NavHeader navigation={this.props.navigation} showBack />
        {
          this.state.dataLoaded
          ?
            <ScrollView>
              <View style={Styles.searchBar}>
                <View style={Styles.inputWithIconContainer}>
                  <Icon name="search" size={20} color="#797777" />
                  <TextInput
                    style={Styles.searchInput}
                    value={this.state.searchTerm}
                    onChange={e => this.searchItems(e.nativeEvent.text)}
                    onChangeText={text => this.setState({ searchTerm: text })}
                    underlineColorAndroid="transparent"
                    placeholder="Cerca tra i lavori"
                    placeholderTextColor="#797777"
                    returnKeyType="search"
                  />
                </View>
                <TouchableOpacity
                  onPress={() => this.showFilter()}
                  style={Styles.filterIconContainer}
                >
                  <Image source={filterLogo} style={Styles.filterLogo} />
                </TouchableOpacity>
              </View>
              {this.displayFilter()}
              <View style={Styles.titleContianer}>
                <Text style={Styles.title}>
                  {this.state.title}
                </Text>
                <Text style={Styles.count}>{'(' + this.state.productList.length + ')'}</Text>
              </View>
              <FlatList
                extraData={this.state}
                removeClippedSubviews
                data={this.state.productList}
                renderItem={({ item }) => this.renderProductCard(item)}
              />
              <View style={Styles.buttonView}>
                <TouchableOpacity onPress={() => this.addNewItem()} style={Styles.button}>
                  <Text style={Styles.buttonLabel}>
                    Carica pi√π elementi
                  </Text>
                </TouchableOpacity>
              </View>
            </ScrollView>
          :
            <View style={Styles.loaderView}>
              <ActivityIndicator size="large" color="#5F7070" />
            </View>
      }
      </SafeAreaView>
    );
  }
}
 
VetrinaLavori.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
