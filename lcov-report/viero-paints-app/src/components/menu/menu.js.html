/**
 * Menu Component
 * created @ 17/10/2018
 * This files contains the code for the Menu Component
 */
 
import React from 'react';
import Icon from 'react-native-vector-icons/FontAwesome';
import PropTypes from 'prop-types';
import {
  SafeAreaView,
  View,
  Text,
  Dimensions,
  TouchableOpacity,
} from 'react-native';
import { getLocalData } from '../../controller/apicontroller';
import NavHeader from '../navheader/navheader';
import MenuItems from './menuitems';
import Styles from './styles';
 
const { width } = Dimensions.get('window');
 
export default class Menu extends React.Component {
 
  constructor() {
    super();
    this.state = {
      name: '',
      userData: '',
    };
  }
 
  componentDidMount() {
    getLocalData('userData').then(
      userData => this.loadCustomerData(userData),
    );
  }
 
  loadCustomerData(data) {
    const userData = JSON.parse(data);
    const name = userData.firstname + ' ' + userData.lastname;
    this.setState({ name: name, userData: userData });
  }
 
  showScreen(key) {
    let data = '';
    if (key === 'Profile') {
      data = this.state.userData;
    }
    if (key === 'VETRINA LAVORI') {
      this.props.navigation.navigate('VetrinaLavori', { data: data });
    } else {
      const screen = key.toLowerCase().replace(/\w/, c => c.toUpperCase());
      this.props.navigation.navigate(screen, { data: data });
    }
  }
 
  render() {
    return (
      <SafeAreaView style={Styles.container} testID={'SideMenu'}>
        <NavHeader navigation={this.props.navigation} showBack />
        <View style={Styles.menuView}>
          {
          MenuItems.map(item =>
            <TouchableOpacity
              testID={item.testID}
              accessibilityLabel={item.testID}
              onPress={() => this.showScreen(item.name)}
              style={Styles.menuButton}
            >
              <Text style={Styles.menuLabel}>{item.name}</Text>
            </TouchableOpacity>
           )
         }
          <View style={Styles.footerView}>
            <TouchableOpacity
              testID={'ProfileButton'}
              accessibilityLabel={'ProfileButton'}
              onPress={() => this.showScreen('Profile')}
              style={Styles.settingsView}
            >
              <Icon name="cog" size={width * 0.04} color="rgba(255, 255, 255, 0.6)" />
              <Text style={Styles.settingsLabel}>
               Modifica profilio
              </Text>
            </TouchableOpacity>
            <Text style={Styles.name}>
              {this.state.name}
            </Text>
          </View>
        </View>
      </SafeAreaView>
    );
  }
}
 
Menu.propTypes = {
  navigation: PropTypes.shape({
    navigate: PropTypes.func.isRequired,
    goBack: PropTypes.func.isRequired,
  }).isRequired,
};
