    getToken().then(token => this.loadFilters(token));

      data => this.processFilters(data)
    );
  }
 
  processFilters(data) {
    const filters = data.children_data[0].children_data;
    this.setState({ filters: filters });
  }
 
  showFilter() {
    if (this.state.showFilter) {
      this.setState({ showFilter: false });
    } else {
      this.setState({ showFilter: true });
    }
  }

          filterItems={this.filterItems}

  filterItems(item) {
    const title = item.name;
    this.setState({ title: title, dataLoaded: false, showFilter: false });
    const id = item.id;
    let url = '';
    if ((item.name === 'Materiale Marketing') || (item.name === 'Abbigliamento & Strumenti')) {
      url = PRODUCTTI_URL_HEAD + id + PRODUCTTI_SIMPLE_URL_TAIL;
    } else {
      url = PRODUCTTI_URL_HEAD + id + PRODUCTTI_URL_TAIL;
    }
    getToken().then(token => this.loadProducts(token, url));
  }
 
  showProductDetails(item) {
    this.props.navigation.navigate('ProductDetails', { data: item });
  }
 
  renderProductCard(item) {
    const imageIndex = item['custom_attributes'].findIndex(x => x.attribute_code === 'thumbnail');
    const descriptionIndex = item['custom_attributes'].findIndex(x => x.attribute_code === 'short_description');
    return (
      <TouchableOpacity
        onPress={() => this.showProductDetails(item)}
        style={Styles.productView}
      >
        {
           imageIndex >= 0
          ?
            <Image source={{ uri: MEDIA_URL + item.custom_attributes[imageIndex]['value'] }} style={Styles.productImage} />
          :
            <Image source={dummyImage} style={Styles.productImage} />
        }
        <View style={Styles.productTitleView}>
          <Text numberOfLines={1} style={Styles.productTitle}>{item.name} </Text>
          <Text numberOfLines={1} style={Styles.productDescription}>{item.custom_attributes[descriptionIndex]['value'].replace(/<\/?[^>]+(>|$)/g, '')}</Text>
        </View>
      </TouchableOpacity>
    );
